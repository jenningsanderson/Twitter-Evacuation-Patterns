def sheltered_in_place?
        during_storm_p = cluster_as_point(during_storm_cluster)
        unless during_storm_p.nil? or base_cluster_point.nil?
                return during_storm_p.distance(base_cluster_point) < 500
        else
                return nil
        end
end
